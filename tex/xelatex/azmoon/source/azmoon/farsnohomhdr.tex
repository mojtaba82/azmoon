\newcommand{\dotrule}[1]{%
	\parbox[t]{#1}{\dotfill}%
}
\newcommand\nameordot[2][]{%
	\edef\namearg{#2}%
	\edef\lenarg{#1}%
	\ifx\namearg\empty%
		\ifx\lenarg\empty%
			\dotfill%
		\else%
			\dotrule{#1}%
		\fi%
	\else%
		\namearg
	\fi%
}
